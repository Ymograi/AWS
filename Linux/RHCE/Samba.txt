yum install -y samba samba-client;
mkdir /sambashare;
getsebool -a | grep samba_export;
getsebool -a | grep samba_share_nfs;
setsebool -P samba_export_all_ro=1 samba_export_all_rw=1 samba_share_nfs=1;
semanage fonctext -at samba_share_t "/sambashare(/.*)?";
restorecon /sambashare/;
ll -Zd /sambashare/;
firewall-cmd --permanent --add-service=samba;
firewall-cmd --reload;
cd /etc/samba;
mv smb.conf smb.conf.org.backup;
