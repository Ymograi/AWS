
yum install -y openssl elinks;sleep 3;c

cd /etc/pki/tls/certs/ ;sleep 3;c
openssl genpkey -algorithm  rsa -pkeyopt rsa_keygen_bits:2048 -out server1.example.com.key;sleep 3;c
ll;sleep 3;c
cat server1.example.com.key;sleep 3;c
openssl req -new -key server1.example.com.key -out server1.example.com.csr;sleep 3;c
cat server1.example.com.csr;sleep 3;c
openssl x509 -req -days 365 -signkey server1.example.com.key -in server1.example.com.csr -out server1.example.com.crt;sleep 3;c
openssl s_client -connect localhost:443 -state;sleep 3;c

echo '<VirutalHost *:443>
 DocumentRoot "/var/www/html/ourexample"
 ServerName server1.example.com:443
 SSLCertificateFile /etc/pki/tls/certs/server1.example.com.crt
 SSLCertificateKeyFile /etc/pki/tls/certs/server1.example.com.key
</VirutalHost>' >> /etc/httpd/conf.d/ssl.conf;sleep 3;c

mkdir /var/www/html/ourexample;sleep 3;c
cd /var/www/html/ourexample;sleep 3;c
echo "This is a SecureWebsite Now" > index.html;sleep 3;c
restorecon -Rv /var/www/html/ourexample;sleep 3;c
